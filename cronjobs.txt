#!/bin/bash
# Trading Bot Data Collection Cron Job
# Runs every hour to build historical database

# Run data collector every hour
0 * * * * /usr/bin/python3 /home/clawdbot/polymarket-bot/src/data_collector.py --once >> /home/clawdbot/polymarket-bot/logs/cron.log 2>&1

# Weekly maintenance: health checks, tests, auto-fix, commit/push (Sundays 3am UTC)
0 3 * * 0 /usr/bin/python3 /home/clawdbot/polymarket-bot/maintenance.py >> /home/clawdbot/polymarket-bot/logs/maintenance.log 2>&1
